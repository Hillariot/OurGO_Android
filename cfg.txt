name: Release iOS

on:
  push:
    tags:
      - "version_ios*.*.*"

permissions:
  contents: read

jobs:
  build-ios:
    runs-on: macos-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup .NET
        uses: actions/setup-dotnet@v4
        with:
          dotnet-version: "9.0.x"

      - name: dotnet info
        run: |
          dotnet --list-sdks
          dotnet --info

      - name: Install MAUI workloads
        run: dotnet workload install maui-ios

      # ---------- Install certificate ----------
      - name: Install Apple certificate
        env:
          IOS_CERT_BASE64: ${{ secrets.IOS_CERT_BASE64 }}
          IOS_CERT_PASSWORD: ${{ secrets.IOS_CERT_PASSWORD }}
        run: |
          echo $IOS_CERT_BASE64 | base64 --decode > ios_cert.p12
          security create-keychain -p "" build.keychain
          security import ios_cert.p12 -k build.keychain -P "$IOS_CERT_PASSWORD" -T /usr/bin/codesign
          security list-keychains -s build.keychain

      # ---------- Install provisioning profile ----------
      - name: Install provisioning profile
        env:
          IOS_PROFILE_BASE64: ${{ secrets.IOS_PROFILE_BASE64 }}
        run: |
          mkdir -p ~/Library/MobileDevice/Provisioning\ Profiles
          echo $IOS_PROFILE_BASE64 | base64 --decode > profile.mobileprovision
          cp profile.mobileprovision ~/Library/MobileDevice/Provisioning\ Profiles/

      # ---------- Build IPA ----------
      - name: Build iOS IPA
        run: |
          dotnet publish ./OurGO/OurGO/OurGO.csproj \
            -c Release \
            -f net10.0-ios \
            -p:BuildIpa=true \
            -o ./artifacts/ios

      # ---------- Collect artifacts ----------
      - name: Collect IPA
        run: |
          mkdir -p artifacts/releases/ios
          cp artifacts/ios/*.ipa artifacts/releases/ios/
          ls artifacts/releases/ios

      # ---------- Create release ----------
      - name: Create release if missing
        env:
          GH_TOKEN: ${{ secrets.IOS_TOKEN }}
        run: |
          gh release view $GITHUB_REF_NAME --repo "Hillariot/OurGO_iOS" || \
          gh release create $GITHUB_REF_NAME \
            --repo "Hillariot/OurGO_iOS" \
            --title $GITHUB_REF_NAME \
            --notes "iOS release $GITHUB_REF_NAME"

      # ---------- Upload IPA ----------
      - name: Upload IPA
        env:
          GH_TOKEN: ${{ secrets.IOS_TOKEN }}
        run: |
          gh release upload Hillariot/OurGO_Android \
            --clobber \
            --repo "Hillariot/OurGO_iOS"
